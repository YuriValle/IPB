SUB mostraMenu(): VOID
INICIO
	ESCREVER("1 - Soma");
	ESCREVER("2 - Subtracao");
	ESCREVER("3 - Multiplicacao");
	ESCREVER("4 - Divisao");

SUB perguntaNumeros(): VOID
	ESCREVER("Digite o primeiro valor.");
	LER(primeiroNumero);
	ESCREVER("Digite o segundo valor.");
	LER(segundoNumero);
}
SUB soma(): INTEIRO
INICIO
	RETORNAR a + b;
FIM
SUB subtracao(): INTEIRO
INICIO
	RETORNAR a - b;
FIM
SUB multiplicacao(): INTEIRO
INICIO
	RETORNAR a * b;
FIM
SUB divisao(): REAL
INICIO
	RETORNAR a / b;
FIM
